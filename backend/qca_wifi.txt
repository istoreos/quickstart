config wifi-device 'wifi0'
        option type 'qcawificfg80211'
		# 默认auto 
        option channel 'auto'
        option macaddr '94:83:c4:ad:4b:32'
		# 默认 beg， 默认为 11b/g/n/ax/be
        option hwmode '11beg'
		# 选择为 11ax/be 时  添加  option require_mode 'ax'
        option disabled '0'
        option country 'CN'
		# 等级 最大(30)、高(21)、中(15)、低(9)   
        option txpower '9'
		# 默认 1， 开启是1，关闭是0
        option random_bssid '1'
        option band '2g'
		# 默认 HT40, 选择为 20/40 时为HT40, 选择40时也为HT40, 选择20时为HT20
        option htmode 'HT40'
        option legacy_rates '0'
		# 默认配置文件不存在， 选择 htmode 为 单 40 时， 添加 option noscan '1' ，后续切换为20 和 20/40 时，  option noscan '0'
		option noscan '1'

config wifi-iface 'wifi2g'
        option device 'wifi0'
        option network 'lan'
        option mode 'ap'
        option ssid 'GL-BE3600-b30'
        option encryption 'psk2+ccmp'
		# 选择 random_bssid 为 1 的情况下，macaddr 是随机生成
		# 选择 random_bssid 为 1 的情况下，macaddr 和 factory_macaddr 是一样的。
		# 可以理解 factory_macaddr 是每个wifi接口的物理mac, macaddr 是 每个wifi接口的虚拟mac
        option macaddr '0E:33:52:FB:8F:F2'
        option factory_macaddr '94:83:c4:ad:4b:32'
        option key '35P865ZZWM'
        option wds '1'
		# AP隔离功能， guest 默认开启，普通wifi 默认关闭，内部网络中可以开启 AP隔离
        option isolate '0'
		# 默认为0， 隐藏时为1
        option hidden '0'
        option ifname 'wlan0'
        option ieee80211k '1'
        option bss_transition '1'
        option sae '0'
		# 页面单独 htmode 40 时，添加  option disablecoext '1' ，默认不存在，其他选项情况下删除该配置项。
		option disablecoext '1'
